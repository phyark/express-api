### 目录结构
```

│   └── www                   # 启动文件
├── logs                      # 日志文件
├── src
│   ├── config                # 配置文件
│   ├── controllers           # 业务层代码
│   ├── middleware            # 中间件
│   ├── models                # 数据表模型
│   ├── public                # 静态资源
│   ├── routes                # 路由文件
│   ├── services              # 数据层
│   ├── utils                 # 工具类

├── init.sql                  # mysql 数据库初始化
├── ecosystem.config.js       # pm2 配置文件
```
